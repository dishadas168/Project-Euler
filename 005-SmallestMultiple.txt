int gcd(int a, int b)
{
    if (b == 0)
        return a;
    return gcd(b, a % b);
}

int smallestMultiple(int t){
    // Initialize result
    int ans = 1;

    // after i'th iteration,
    for (int i = 2; i <= t; i++)
        ans = (((i * ans)) /
                (gcd(i, ans)));
 
    return ans;
}
