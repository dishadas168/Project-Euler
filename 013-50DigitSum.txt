string getSum(vector<string> nums){
    string ans = "";
    int col_sum = 0, ones = 0;
    for(int i= 49; i >=0; i--){
        for(int j = 0; j< nums.size(); j++)
            col_sum += nums[j][i]-'0';
        ones = col_sum % 10;
        col_sum = col_sum / 10;
        ans = to_string(ones) + ans;
    }
    ans = to_string(col_sum) + ans;

    return ans.substr(0,10); 
}
