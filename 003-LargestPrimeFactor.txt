long long largestPF(long long num){
    long long ans = num;
    long long i=2, max=1;
  
    while(ans%2 == 0){
            ans = ans/2;
            max = 2;
    }
    
    i = 3;
    while(i <= sqrt(ans)){
        while(ans%i == 0){
            ans = ans/i;
            if(i > max) max = i;
        }      
        i+=2;
    }
    return std::max(max, ans);
}
